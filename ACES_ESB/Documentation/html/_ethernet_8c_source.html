238 void waitMS(uint16_t msec)
239 {
240  // This function utilizes timer 0 and a sequence of delay loops to delay for the desired time
241  TCNT0 = 5;
242  // begin the timer
243  TCCR0B |= (1 << CS01) | (1 << CS00); // this will start the timer with a prescalar of 64
244  for (uint16_t i = 0; i < msec; i++){
245  while(bit_is_clear(TIFR0, TOV0));
246  TIFR0 |= (1 << TOV0); // Clear the overflow flag by writing a 1 to it
247  }
248  assign_bit(&TCCR0B, CS01, 0);
249  assign_bit(&TCCR0B, CS00, 0); // Turn off the timer
250 }

256 void InitEthernet(void)
257 {
258  // First perform a system reset. This will ensure register start with their expected values
259  readBasic(SOFT_RESET,SOFT_RESET); // normally read returns something but this will serve as a write with fewer operations
260  waitMS(50); // give the chip time to restart
261 
263  nextPacketPtr = RXSTART_INIT;
264 
265  // Rx start
266  RegisterWrite(ERXSTL, RXSTART_INIT); // Low byte first and then high byte
267  RegisterWrite(ERXSTH, RXSTART_INIT >> 8);
268 
269  // Now set the current receive pointer address
270  RegisterWrite(ERDPTL, RXSTART_INIT);
271  RegisterWrite(ERDPTH, RXSTART_INIT >> 8); // Again, low byte first
272 
273  // RX end
274  RegisterWrite(ERXNDL, RXSTOP_INIT & 0xFF); // doing the 0xFF will trim off the higher byte
275  RegisterWrite(ERXNDH, RXSTOP_INIT >> 8);
276 
277  // TX Start
278  RegisterWrite(ETXSTL, TXSTART_INIT & 0xFF);
279  RegisterWrite(ETXSTH, TXSTART_INIT >> 8);
280 
281  // TX end
282  RegisterWrite(ETXNDL, TXSTOP_INIT & 0xFF);
283  RegisterWrite(ETXNDH, TXSTOP_INIT >> 8);
284 
285  // Now configure the packet filters
286  // Everything will be filtered with Unicast
287  RegisterWrite(ERXFCON, ERXFCON_ANDOR); // have no filters for now
288  //RegisterWrite(ERXFCON, ERXFCON_UCEN | ERXFCON_ANDOR);
289  // The above line will ignore all packets which don't have the destination address
290  // matching this device's MAC address
291 
292  // Now enable the MAC and other features
293  // This will enable the MAC to receive packets
294  // allow MAC to transmit pause control frame
295  // and inhibit transmission when pause control frames are received
296  RegisterWrite(MACON1, MACON1_MARXEN | MACON1_TXPAUS | MACON1_RXPAUS);
297  RegisterWrite(MACON2, 0); // this will cause the MAC to enter "normal operation" as per the data sheet p61

302  RegisterWrite(MAIPGH, 0x0C); // Also what the data sheet recommended
303

307  RegisterWrite(MAADR4, ESB_mac[1]);

363 uint16_t packetRecieve(uint8_t max_length, uint8_t *packets)
364 {
365  uint16_t rxstat;
366  uint16_t len;
367 
368  // Check to see if a packet has been received, if not, then return
369  if (!RegisterRead(EPKTCNT))
370  return(0);
371 
372  // Set the read pointer to the start of the received packet
373  RegisterWrite(ERDPTL, nextPacketPtr);
374  RegisterWrite(ERDPTH, nextPacketPtr >> 8);
375 
376  // Read the next packet pointer (see page 43 in data sheet)
377  nextPacketPtr = readBasic(READ_BUF_MEM, 0);
378  nextPacketPtr |= readBasic(READ_BUF_MEM, 0) << 8;
379  // Reading from the memory buffer automatically increments to the next byte
380 
381  // read the packet length
382  len = readBasic(READ_BUF_MEM, 0);
383  len |= readBasic(READ_BUF_MEM, 0) << 8;
384  // in the example, 4 is subtracted from this but I don't have CRC checking implemented since I don't think it is necessary
385 
386  // read the receive status
387  rxstat = readBasic(READ_BUF_MEM, 0);
388  rxstat |= readBasic(READ_BUF_MEM, 0) << 8;
389 
390  // limit the receive length
391  if (len > max_length)
392  len = max_length;
393 
394  // check for symbol errors
395  if (!(rxstat & 0x80)){
396  // this is an invalid packet
397  len = 0;
398  }
399  else{
400  readBuffer(len, packets);
401  }
402 
403  // move the RX pointer to the start of the next received packet
404  // This frees the memory we just read out
405  RegisterWrite(ERXRDPTL, nextPacketPtr);
406  RegisterWrite(ERXRDPTH, nextPacketPtr >> 8);
407 
408  // decrement the packet counter indicate we are done with this packet
409  writeBasic(BIT_FIELD_SET, ECON2, ECON2_PKTDEC);
410  return(len);
411 }
