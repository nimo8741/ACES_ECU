16 uint8_t readBasic(uint8_t op, uint8_t address)
17 {
18  CSACTIVE;
19  // Now tell the chip that I am trying to read
20  SPDR = op | (address & ADDR_MSK); // this will make it so only the last 5 bits are preserved of the address
21  waitSPI(); // wait for the SPI interrupt flag to be set
22 
23  // read the data
24  SPDR = 0;
25  waitSPI();
26 
27  // If the address is a MAC or MII register then the byte which will be picked up by SPRD
28  // will be dummy data
29  if (address & 0x80){
30  SPDR = 0;
31  waitSPI();
32  }
33  CSPASSIVE; // raise the CS line
34  return SPDR;

50 {
51  // First pull the CS line low
52  CSACTIVE;
53  // Now issue the read buffer memory command
54  SPDR = READ_BUF_MEM;
55  waitSPI(); // Now wait for the message to be sent
56 
57  while(len)
58  {
59  len--;
60  SPDR = 0; // Just give the register some data so that it begins the transmission
61  waitSPI();
62  *data = SPDR;

64  }

66  CSPASSIVE; // Raise the CS line to stop the transmission

174 void waitMS(uint16_t msec)
175 {
176  // This function utilizes timer 0 and a sequence of delay loops to delay for the desired time
177  TCNT0 = 5;
178  // begin the timer
179  TCCR0B |= (1 << CS01) | (1 << CS00); // this will start the timer with a prescalar of 64
180  for (uint16_t i = 0; i < msec; i++){
181  while(bit_is_clear(TIFR0, TOV0));
182  TIFR0 |= (1 << TOV0); // Clear the overflow flag by writing a 1 to it
183  }
184  assign_bit(&TCCR0B, CS01, 0);
185  assign_bit(&TCCR0B, CS00, 0); // Turn off the timer
186 }

188 void InitEthernet(void)
189 {
190  // First perform a system reset. This will ensure register start with their expected values
191  readBasic(SOFT_RESET,SOFT_RESET); // normally read returns something but this will serve as a write with fewer operations
192  waitMS(50); // give the chip time to restart
193 
195  nextPacketPtr = RXSTART_INIT;
196 
197  // Rx start
198  RegisterWrite(ERXSTL, RXSTART_INIT & 0xFF); // Low byte first and then high byte
199  RegisterWrite(ERXSTH, RXSTART_INIT >> 8);
200 
201  // Now set the current receive pointer address
202  RegisterWrite(ERXRDPTL, RXSTART_INIT & 0xFF);
203  RegisterWrite(ERXRDPTH, RXSTART_INIT >> 8); // Again, low byte first
204 
205  // RX end
206  RegisterWrite(ERXNDL, RXSTOP_INIT & 0xFF); // doing the 0xFF will trim off the higher byte
207  RegisterWrite(ERXNDH, RXSTOP_INIT >> 8);
208 
209  // TX Start
210  RegisterWrite(ETXSTL, TXSTART_INIT & 0xFF);
211  RegisterWrite(ETXSTH, TXSTART_INIT >> 8);
212 
213  // TX end
214  RegisterWrite(ETXNDL, TXSTOP_INIT & 0xFF);
215  RegisterWrite(ETXNDH, TXSTOP_INIT >> 8);
216 
217  // Now configure the packet filters
218  // Everything will be filtered with Unicast
219 
220  //RegisterWrite(ERXFCON, ERXFCON_UCEN | ERXFCON_ANDOR);
221 
222  // The above line will ignore all packets which don't have the destination address
223  // matching this device's MAC address
224 
225  // Now enable the MAC and other features
226  // This will enable the MAC to receive packets
227  // allow MAC to transmit pause control frame
228  // and inhibit transmission when pause control frames are received
229  RegisterWrite(MACON1, MACON1_MARXEN | MACON1_TXPAUS | MACON1_RXPAUS);
230  RegisterWrite(MACON2, 0); // this will cause the MAC to enter "normal operation" as per the data sheet p61
231 
232  // Now do Inter-Packet gap (not back-to-back)
233  RegisterWrite(MAIPGL, 0x12); // This was what the data sheet recommended (p34)
234  RegisterWrite(MAIPGH, 0x0C); // Also what the data sheet recommended
235 
236  // Set the maximum packet size
237  RegisterWrite(MAMXFLL, MAX_FRAMELEN & 0xFF);
238  RegisterWrite(MAMXFLH, MAX_FRAMELEN >> 8);
239 
240 
241  // Now write the MAC Address to the device
242  // Note, the ENC28J60 is byte backwards, at least that is the consensus on the Internet
243  RegisterWrite(MAADR5, ECU_mac[0]);
244  RegisterWrite(MAADR4, ECU_mac[1]);
245  RegisterWrite(MAADR3, ECU_mac[2]);
246  RegisterWrite(MAADR2, ECU_mac[3]);
247  RegisterWrite(MAADR1, ECU_mac[4]);
248  RegisterWrite(MAADR0, ECU_mac[5]);
249 
250  // Now prevent loop back of transmitted frames
251  PhyWrite(PHCON2, PHCON2_HDLDIS | PHCON2_FRCLINK);
252 
253  // enable packet reception
254  writeBasic(BIT_FIELD_SET, ECON1, ECON1_RXEN);
255 
256  // now enable the interrupt flags
257  writeBasic(BIT_FIELD_SET, EIE, EIE_PKTIE | EIE_TXERIE | EIE_RXERIE);
258  // The above line will enable the interrupt flags for Receive Packet Pending, Transmit, Transmit error, and receive packet error
259 }

299 uint16_t packetRecieve(uint8_t max_len, uint8_t *packets)
300 {
301  uint16_t rxstat;
302  uint16_t len;
303 
304  // Check to see if a packet has been received, if not, then return
305  if (!RegisterRead(EPKTCNT))
306  return(0);
307 
308  // Set the read pointer to the start of the received packet
309  RegisterWrite(ERDPTL, nextPacketPtr);
310  RegisterWrite(ERDPTH, nextPacketPtr >> 8);
311 
312  // Read the next packet pointer (see page 43 in data sheet)
313  nextPacketPtr = readBasic(READ_BUF_MEM, 0);
314  nextPacketPtr |= readBasic(READ_BUF_MEM, 0) << 8;
315  // Reading from the memory buffer automatically increments to the next byte
316 
317  // read the packet length
318  len = readBasic(READ_BUF_MEM, 0);
319  len |= readBasic(READ_BUF_MEM, 0) << 8;
320  // in the example, 4 is subtracted from this but I don't have CRC checking implemented since I don't think it is necessary
321 
322  // read the receive status
323  rxstat = readBasic(READ_BUF_MEM, 0);
324  rxstat |= readBasic(READ_BUF_MEM, 0) << 8;
325 
326  // limit the receive length
327  if (len > max_len)
328  len = max_len;
329 
330  // check for symbol errors
331  if (!(rxstat & 0x80)){
332  // this is an invalid packet
333  len = 0;
334  }
335  else{
336  readBuffer(len, packets);
337  }
338 
339  // move the RX pointer to the start of the next received packet
340  // This frees the memory we just read out
341  RegisterWrite(ERXRDPTL, nextPacketPtr);
342  RegisterWrite(ERXRDPTH, nextPacketPtr >> 8);
343 
344  // decrement the packet counter indicate we are done with this packet
345  writeBasic(BIT_FIELD_SET, ECON2, ECON2_PKTDEC);
346  return(len);
347 }
